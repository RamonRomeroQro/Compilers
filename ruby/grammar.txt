PROGRAM -> BLOCK           

BLOCK   -> T STATEMENT T BLOCK	        o
	    | T STATEMENT T 

T -> T newline         

STATEMENT -> OUTPUT | ASIGN | LOOP | CONDITIONALIFELSE | STATEMENT  | INPUT

INPUT -> gets.chomp
       | gets.chomp.to_i   


OUTPUT -> puts STRING           
        | print INTEGER
        | print BOOLEAN
        | print STRING

ASIGN -> VARIABLEID EQUALS EXPR
        | CONSTANTID EQUALS INTEGER
        | CONSTANTID EQUALS "TRUE" 
        | CONSTANTID EQUALS "FALSE" 
        | CONSTANTID EQUALS STRING
        

STRING  -> STRING
        | STRING PLUS STRING


EVALUATION -> EXPR less EXPR
	    | EXPR equals equals EXPR
	    | EXPR great EXPR
	    | EXPR great equals EXPR
	    | EXPR less equals EXPR
        | NOT EVALUATION
        | EXPR and EXPR
        | EXPR or EXPR

EXPR -> EXPR plus EXPR
	| EXPR minus EXPR
	| EXPR times EXPR
	| EXPR divide EXPR
	| EXPR modulo EXPR
    | EVALUATION
	| INTEGER
    | STRING
    | BOOLEAN
    | VARIABLEID


LOOP  ->  while EVALUATION do BLOCK end


CONDITIONALIFELSE -> if EVALUATION BLOCK else BLOCK end
                    | if EVALUATION BLOCK end
